옶션 : 
git reset --soft HEAD~1
--SOFT 		->커밋 내용은 사라지는데 내용을 임시저장 상태롤 되돌린다
git reset --mixed HEAD~1
--MIXED	->임시 저장도 취소한다. 작업 공간에는 남겨둠
git reset --hard HEAD~1
--HARD		->임시 저장도 취소 , 작업 공간에서도 삭제
